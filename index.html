Recipe 1: fibonacci generator

            
                #!/usr/bin/env python
                def fibo_generator(N=10000):
                    n1 = 0
                    n2 = 1
                    while n2 <= N:
                        n2, n1 = n2 + n1, n2
                        if n2 < N:
                            yield n2
                    raise StopIteration

                if __name__ == '__main__':
                    for i in fibo_generator(N=1000):
                        print(i)
